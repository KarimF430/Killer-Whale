            {/* Message Actions */}
            <div className="message-actions">
              <button className="action-btn">

              </button>
              <button className="action-btn">
                <ThumbsDown size={14} />
              </button>
            </div>
          </div>
        </div>
      </div>
    ))}
  </div>

  {/* Input */}
  <div className="input-container">
    <div className="input-wrapper">
      <textarea 
        className="input-box"
        placeholder="Ask anything..."
        value={input}

  </div>
